MapReduce&yarn学习笔记

大数据
发布日期:   2021-01-26

一、宏观认识

  简单认识：split-逻辑上的数据分片，控制map并行度，数据以一条为单位经过map方法映射成KV，相同的K为一组，同一组数据调用一次reduce方法，在方法内迭代计算形成最后的数据集；map后的同一组数据不能被分配到不同的reduce分区，否则会破坏同一组数据计算的准确性。

  • 切片会格式化记录，以记录为单位调用map方法
  • map输出映射为KV，接着由K计算出分区P，KVP
  • map任务的输出数据通过buffer缓冲区异写，按分区P，K排序输出为内部有序的小文件，最终归并成一个分区有序的文件存储在本地文件系统中
  • reduce端拉取文件并归并排序，归并到两个文件就可以同时进行reduce计算，因为有迭代式模式的支持

二、实现

1、计算向数据移动

  • hdfs暴露数据的位置

    资源管理

    任务调度

2、角色

  • Client

    • 做计算资源规划

    • 根据每次的计算数据，从NN上获取Block元数据并计算split切片列表，split包含逻辑上的偏移量、节点信息

    • 生成计算程序所需的配置文件，并将split列表、配置文件、计算程序jar上传到HDFS上

    • 调用JobTracker，通过启动相应的计算程序，并告知所需文件在HDFS上的位置

  • JobTracker

    • 资源管理

    • 任务调度

    • 收到计算程序后，从hdfs获取split列表，根据TaskTracker上报的资源信息，确定每一个split该由哪个节点执行

    • 执行任务信息会在TaskTracker做心跳连接时告知

  • TaskTracker

    • 任务管理

    • 资源汇报

    • 向JobTracker做心跳连接时取回任务信息

    • 从hdfs上下载任务所需资源

    • 最终启动任务描述中的map任务或者reduce任务

      3、问题

JobTracker单点故障、压力过大，即承担资源管理又要实现任务调度，两者耦合在一起，不能复用，多个JobTracker之间隔离，对彼此的资源控制无感知，无法做到宏观调控

三、yarn

  统一资源调度层

  • ResourceManager

    • client申请AppMaster，RM选择一台不忙的节点通知NM启动一个Container，反射一个AppMster-

    • 启动AppMaster，从hdfs上拉取任务信息，向RM申请所需资源

    • RM根据资源情况和APP申请的资源信息规划任务信息，并通知NM来启动container

    • container启动后会反向注册到已启动的AppMaster

    • AppMaster最终将任务Task发给containner

    • container会发射响应的Task类为对象，调用方法执行任务

  • NodeManager

    • 向RM汇报资源情况
    • 根据RM指令，启动相应的container

        System.setProperty("HADOOP_USER_NAME","wps");
        Configuration conf = new Configuration(true);
        conf.set("mapreduce.app-submission.cross-platform","true");//hadoop jar放肆不要设置，windows环境执行必须设置
        /**
         * 源码example
         * // Create a new Job
         *  *     Job job = Job.getInstance();
         *  *     job.setJarByClass(MyJob.class);
         *  *
         *  *     // Specify various job-specific parameters
         *  *     job.setJobName("myjob");
         *  *
         *  *     job.setInputPath(new Path("in")); 有问题
         *  *     job.setOutputPath(new Path("out"));
         *  *
         *  *     job.setMapperClass(MyJob.MyMapper.class);
         *  *     job.setReducerClass(MyJob.MyReducer.class);
         *  *
         *  *     // Submit the job, then poll for progress until the job is complete
         *  *     job.waitForCompletion(true);
         */
        Job job = Job.getInstance(conf);
        job.setJarByClass(WordCount.class);
        job.setJobName("my-wordcount");
        job.setJar("D:\\coding\\test\\target\\test-1.0-SNAPSHOT.jar");//hadoop jar的方式不需要，此处远程提交，hadoop集群才能找到类并反射执行方法

        TextInputFormat.addInputPath(job,new Path("/user/wps/t1"));
        Path outPath = new Path("/user/wps/my_wordcount_out");
        if(outPath.getFileSystem(conf).exists(outPath)){
            outPath.getFileSystem(conf).delete(outPath,true);
        }
        TextOutputFormat.setOutputPath(job,outPath);

        //必须设置，TextOutputFormat的默认输出key为LongWritable，value为Text
        job.setOutputKeyClass(Text.class);
        job.setOutputValueClass(IntWritable.class);

        //设置mapper类和reducer类
        job.setMapperClass(WordCountMapper.class);
        job.setReducerClass(WoerdCountReducer.class);

        job.waitForCompletion(true);
    }
}
  • mapper
package com.wps.mr;

import org.apache.hadoop.io.IntWritable;
import org.apache.hadoop.io.Text;
import org.apache.hadoop.mapreduce.Mapper;

import java.io.IOException;
import java.util.StringTokenizer;

/**
 * run源码：
 * public void run(Context context) throws IOException, InterruptedException {
 *     setup(context);
 *     try {
 *       while (context.nextKeyValue()) {
 *         map(context.getCurrentKey(), context.getCurrentValue(), context);
 *       }
 *     } finally {
 *       cleanup(context);
 *     }
 *   }
 */

public class WordCountMapper extends Mapper<Object, Text,Text, IntWritable> {

    private final IntWritable one = new IntWritable(1);
    private Text word = new Text();

    //单次map任务
    @Override
    protected void map(Object key, Text value, Context context) throws IOException, InterruptedException {
        //分割字符，以字符为key，value为1
        StringTokenizer tokenizer = new StringTokenizer(value.toString());
        while(tokenizer.hasMoreTokens()){
            word.set(tokenizer.nextToken());
            context.write(word,one);
        }
    }

    //初始化操作
    @Override
    protected void setup(Context context) throws IOException, InterruptedException {
        super.setup(context);
    }

    //清理操作
    @Override
    protected void cleanup(Context context) throws IOException, InterruptedException {
        super.cleanup(context);
    }

    @Override
    public void run(Context context) throws IOException, InterruptedException {
        super.run(context);
    }
}
  • reducer
package com.wps.mr;

import org.apache.hadoop.io.IntWritable;
import org.apache.hadoop.io.Text;
import org.apache.hadoop.mapreduce.Reducer;

import java.io.IOException;

public class WoerdCountReducer extends Reducer<Text, IntWritable,Text,IntWritable> {

    private IntWritable reduceValue = new IntWritable(0);

    //相同key 的一组数据会调用一次reduce
    @Override
    protected void reduce(Text key, Iterable<IntWritable> values, Context context) throws IOException, InterruptedException {
        int sum = 0;
        for (IntWritable v: values) {
            sum += v.get();
        }
        reduceValue.set(sum);//本次计算同一组key 的合计数量value
        context.write(key,reduceValue);
    }
}

//FileInputFormat#getSplits
//计算分片实现方法
public List<InputSplit> getSplits(JobContext job) throws IOException {
    Stopwatch sw = new Stopwatch().start();
    //getFormatMinSplitSize()->return 1,getMinSplitSize(job)->用户配置的				mapreduce.input.fileinputformat.split.minsize大小，取最小，即默认为1
    long minSize = Math.max(getFormatMinSplitSize(), getMinSplitSize(job));
    //用户配置mapreduce.input.fileinputformat.split.maxsize大小，否则取Long.MAX_VALUE
    long maxSize = getMaxSplitSize(job);

    // generate splits
    List<InputSplit> splits = new ArrayList<InputSplit>();
    //获取计算文件信息
    List<FileStatus> files = listStatus(job);
    for (FileStatus file: files) {
      Path path = file.getPath();
      long length = file.getLen();
      if (length != 0) {
        BlockLocation[] blkLocations;
        //获取计算文件的Block信息
        if (file instanceof LocatedFileStatus) {
          blkLocations = ((LocatedFileStatus) file).getBlockLocations();
        } else {
          FileSystem fs = path.getFileSystem(job.getConfiguration());
          blkLocations = fs.getFileBlockLocations(file, 0, length);
        }
        if (isSplitable(job, path)) {
          long blockSize = file.getBlockSize();
          //计算当前计算文件的切片大小，不同文件blockSize大小可能不一样
          // return Math.max(minSize, Math.min(maxSize, blockSize));
          //blockSize和maxSize取最大，再与minSize取最大，maxSize由认为干预，最大取blockSize还是比blockSize大的一个值，不设置则默认最大为blockSize，即切片大小和block大小一致
          long splitSize = computeSplitSize(blockSize, minSize, maxSize);

          long bytesRemaining = length;
          //根据上面计算结果，循环判断当前计算文件剩余大小，分割创建切片对象
          while (((double) bytesRemaining)/splitSize > SPLIT_SLOP) {
            //计算当前切片对应的block下标
            int blkIndex = getBlockIndex(blkLocations, length-bytesRemaining);
            //创建切片对象
            //切片对象包含对应的文件、起始偏移量，大小，逻辑映射的计算文件对应的节点，支撑计算向哪里移动
            splits.add(makeSplit(path, length-bytesRemaining, splitSize,
                        blkLocations[blkIndex].getHosts(),
                        blkLocations[blkIndex].getCachedHosts()));
            bytesRemaining -= splitSize;
          }

          if (bytesRemaining != 0) {
            int blkIndex = getBlockIndex(blkLocations, length-bytesRemaining);
            splits.add(makeSplit(path, length-bytesRemaining, bytesRemaining,
                       blkLocations[blkIndex].getHosts(),
                       blkLocations[blkIndex].getCachedHosts()));
          }
        } else { // not splitable
          splits.add(makeSplit(path, 0, length, blkLocations[0].getHosts(),
                      blkLocations[0].getCachedHosts()));
        }
      } else { 
        //Create empty hosts array for zero length files
        splits.add(makeSplit(path, 0, length, new String[0]));
      }
    }
    // Save the number of input files for metrics/loadgen
    job.getConfiguration().setLong(NUM_INPUT_FILES, files.size());
    sw.stop();
    if (LOG.isDebugEnabled()) {
      LOG.debug("Total # of splits generated by getSplits: " + splits.size()
          + ", TimeTaken: " + sw.elapsedMillis());
    }
    return splits;
  }
